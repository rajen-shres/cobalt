393def _checkout_perform_action(request): 

394 """sub function for the checkout screen, also called directly if only one item in cart""" 

395 

396 # Need to mark the entries that this is covering. The payment call is asynchronous so 

397 # we can't just load all the open basket_entries when we come back or more could have been 

398 # added. 

399 

400 # Get list of event_entry_player records to include. 

401 event_entries = BasketItem.objects.filter(player=request.user).values_list( 

402 "event_entry" 

403 ) 

404 event_entry_players = ( 

405 EventEntryPlayer.objects.filter(event_entry__in=event_entries) 

406 .exclude(payment_status="Paid") 

407 .exclude(payment_status="Free") 

408 .filter(payment_type="my-system-dollars") 

409 .distinct() 

410 ) 

411 # players that are using club pp to pay are pending payments not unpaid 

412 # unpaid would prompt the player to pay for event which is not desired here 

413 event_entry_player_club_pp = ( 

414 EventEntryPlayer.objects.filter(event_entry__in=event_entries) 

415 .filter(payment_type="off-system-pp") 

416 .distinct() 

417 ) 

418 for event_entry in event_entry_player_club_pp: 

419 event_entry.payment_status = "Pending Manual" 

420 event_entry.save() 

421 

422 unique_id = str(uuid.uuid4()) 

423 

424 # map this user (who is paying) to the batch id 

425 PlayerBatchId(player=request.user, batch_id=unique_id).save() 

426 

427 # Get total amount 

428 amount = event_entry_players.aggregate(Sum("entry_fee")) 

429 

430 if amount["entry_fee__sum"]: # something for Payments to do 

431 

432 for event_entry_player in event_entry_players: 

433 event_entry_player.batch_id = unique_id 

434 event_entry_player.save() 

435 

436 # Log it 

437 EventLog( 

438 event=event_entry_player.event_entry.event, 

439 actor=request.user, 

440 action=f"Checkout for event entry {event_entry_player.event_entry.id} for {event_entry_player.player}", 

441 event_entry=event_entry_player.event_entry, 

442 ).save() 

443 

444 return payment_api( 

445 request=request, 

446 member=request.user, 

447 description="Congress Entry", 

448 amount=amount["entry_fee__sum"], 

449 route_code="EVT", 

450 route_payload=unique_id, 

451 url=reverse("events:enter_event_success"), 

452 url_fail=reverse("events:enter_event_payment_fail"), 

453 payment_type="Entry to an event", 

454 book_internals=False, 

455 ) 

456 

457 else: # no payment required go straight to the callback 

458 

459 events_payments_callback("Success", unique_id, None) 

460 messages.success( 

461 request, "Entry successful", extra_tags="cobalt-message-success" 

462 ) 

463 return redirect("events:enter_event_success")

467def checkout(request): 

468 """Checkout view - make payments, get details""" 

469 

470 basket_items = BasketItem.objects.filter(player=request.user)

472 if request.method == "POST":

1403 event = get_object_or_404(Event, pk=event_id) 

1404 congress = get_object_or_404(Congress, pk=congress_id)

1410 ) 

1411 

1412 # Check if entries are open

1417 if event.is_full(): 

1418 return render(request, "events/players/event_full.html", {"event": event}) 

1419 

1420 # Check if draft 

1421 if congress.status != "Published": 

1422 return render(request, "events/players/event_closed.html", {"event": event}) 

1423 

1424 # check if POST. 

1425 # Note: this works a bit differently to most forms in Cobalt. 

1426 # We build our own form and use client side code to validate and 

1427 # modify it. 

1428 # This will work unless someone has 

1429 # deliberately bypassed the client side validation in which case we 

1430 # don't mind failing with an error. 

1431 

1432 if request.method == "POST": 

1433 

1434 # create event_entry 

1435 event_entry = EventEntry() 

1436 event_entry.event = event 

1437 event_entry.primary_entrant = request.user 

1438 event_entry.comment = request.POST.get("comment", None) 

1439 

1440 # see if we got a category 

1441 category = request.POST.get("category", None) 

1442 if category:

1446 answer = request.POST.get("free_format_answer", None) 

1447 if answer: 

1448 event_entry.free_format_answer = answer[:60]

1454 event=event,

1467 # players = {0: request.user} 

1468 # player_payments = {0: request.POST.get("player0_payment")}

1481 if p_id > 3: 

1482 player_payments[p_id] = "Free"

1518 event_entry_player.payment_status = "Pending Manual"

1524 actor=event_entry.primary_entrant, 

1525 action=f"Event entry player {event_entry_player.id} created for {event_entry_player.player}", 

1526 event_entry=event_entry, 

1527 ).save()
